⚙️ Google Apps Script 雲端建表與部署 SOP

本系統不需要任何雲端主機，透過免費的 Google 試算表與 Google Apps Script (GAS) 即可完成資料收集。請依照以下步驟設定您的看板後端：

  1. 建立一個新的 Google 試算表 (Google Sheets)。
  2. 點擊頂部選單的 擴充功能 ＞ Apps Script。
  3. 刪除原本的所有程式碼，複製並貼上底下的黃金防呆 Apps Script 程式碼。
  4. 點擊上方 部署 ＞ 新增部署。
  5. 設定類型為 網頁應用程式：
    • 專案說明：答題快打課堂看板
    • 執行身分：我 (您的 Google 帳戶)
    • 誰有權限存取：所有人 (Anyone) (必須是 Anyone，學生端才能免登入直接 POST 上傳)。
  6. 點擊「部署」，授權存取 Google 帳戶後，複製產生的 「網頁應用程式 URL」。
  7. 將 URL 貼回控制台的「分發與設定」頁面中，點擊儲存即可使用！
// Google Apps Script 答題快打課堂版資料同步腳本 (支援跨域 CORS)
function doGet(e) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("對戰紀錄");
  if (!sheet) {
    return ContentService.createTextOutput(JSON.stringify([]))
      .setMimeType(ContentService.MimeType.JSON)
      .setHeader("Access-Control-Allow-Origin", "*");
  }
  
  var rows = sheet.getDataRange().getValues();
  var data = [];
  var headers = rows[0];
  
  for (var i = 1; i < rows.length; i++) {
    var row = rows[i];
    var record = {};
    for (var j = 0; j < headers.length; j++) {
      record[headers[j]] = row[j];
    }
    data.push(record);
  }
  
  return ContentService.createTextOutput(JSON.stringify(data))
    .setMimeType(ContentService.MimeType.JSON)
    .setHeader("Access-Control-Allow-Origin", "*");
}

function doPost(e) {
  try {
    var params = JSON.parse(e.postData.contents);
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    
    // 自動尋找或建立「對戰紀錄」工作表
    var sheetName = "對戰紀錄";
    var sheet = ss.getSheetByName(sheetName);
    if (!sheet) {
      sheet = ss.insertSheet(sheetName);
      // 自動寫入繁體中文欄位標題，方便教師一覽
      sheet.appendRow(["提交時間", "班級", "座號姓名", "答對題數", "最高連擊", "造成傷害", "錯題數", "錯題詳情"]);
    }
    
    var timestamp = new Date();
    var className = params.className || "未指定";
    var name = params.name || "匿名學生";
    var correct = params.correct !== undefined ? params.correct : 0;
    var bestStreak = params.bestStreak !== undefined ? params.bestStreak : 0;
    var damage = params.damage !== undefined ? params.damage : 0;
    var wrongCount = params.wrongCount !== undefined ? params.wrongCount : 0;
    var wrongList = typeof params.wrongList === "object" ? JSON.stringify(params.wrongList) : (params.wrongList || "[]");
    
    sheet.appendRow([timestamp, className, name, correct, bestStreak, damage, wrongCount, wrongList]);
    
    return ContentService.createTextOutput(JSON.stringify({ status: "success", message: "學生對戰資料已成功同步到 Google Sheet！" }))
      .setMimeType(ContentService.MimeType.JSON)
      .setHeader("Access-Control-Allow-Origin", "*");
  } catch(error) {
    return ContentService.createTextOutput(JSON.stringify({ status: "error", message: error.toString() }))
      .setMimeType(ContentService.MimeType.JSON)
      .setHeader("Access-Control-Allow-Origin", "*");
  }
}

// 處理 CORS 預檢請求
function doOptions(e) {
  return ContentService.createTextOutput("")
    .setHeader("Access-Control-Allow-Origin", "*")
    .setHeader("Access-Control-Allow-Methods", "POST, GET, OPTIONS")
    .setHeader("Access-Control-Allow-Headers", "Content-Type");
}
